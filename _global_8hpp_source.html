290 template<typename ReturnType, typename... Args>
291 ReturnType callPlugins(HookedCall target, HookStep step, bool& skipFunctionCall, Args&&... args) const
292 {
293 using PluginCallType = ReturnType(Args...);
294 constexpr bool ReturnTypeIsVoid = std::is_same_v<ReturnType, void>;
295 using NoVoidReturnType = std::conditional_t<ReturnTypeIsVoid, int, ReturnType>;
296
297 NoVoidReturnType ret {};
298 TRY_HOOK
299 {
300 for (const auto& hook : pluginHooks_[uint(target) * magic_enum::enum_count<HookStep>() + uint(step)])
301 {
302 const auto& plugin = hook.plugin;
303 if (plugin->paused)
304 continue;
305
306 if (plugin->resetCode)
307 *plugin->returnCode = ReturnCode::Default;
308
309 TRY_HOOK
310 {
311 if constexpr (ReturnTypeIsVoid)
312 reinterpret_cast<PluginCallType*>(hook.hookFunction)(std::forward<Args>(args)...);
313 else
314 ret = reinterpret_cast<PluginCallType*>(hook.hookFunction)(std::forward<Args>(args)...);
315 }
316 CATCH_HOOK({ AddLog(LogType::Normal, LogLevel::Err, std::format("Exception in plugin '{}' in {}", plugin->name, __FUNCTION__)); });
317
318 auto code = *plugin->returnCode;
319
320 if ((code & ReturnCode::SkipFunctionCall) != ReturnCode::Default)
321 skipFunctionCall = true;
322
323 if ((code & ReturnCode::SkipPlugins) != ReturnCode::Default)
324 break;
325 }
326 }
327 CATCH_HOOK({ AddLog(LogType::Normal, LogLevel::Err, std::format("Exception {}", __FUNCTION__)); });
328
329 if constexpr (!ReturnTypeIsVoid)
330 return ret;
331 }
332};

334template<typename ReturnType = void, typename... Args>
335auto CallPluginsBefore(HookedCall target, Args&&... args)
336{
337 bool skip = false;
338 if constexpr (std::is_same_v<ReturnType, void>)
339 {
340 PluginManager::i()->callPlugins<void>(target, HookStep::Before, skip, std::forward<Args>(args)...);
341 return skip;
342 }
343 else
344 {
345 ReturnType ret = PluginManager::i()->callPlugins<ReturnType>(target, HookStep::Before, skip, std::forward<Args>(args)...);
346 return std::make_tuple(ret, skip);
347 }
